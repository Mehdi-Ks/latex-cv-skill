  {\fontsize{9}{10}\selectfont\bfseries

}

% ── Job header ─────────────────────────────────────────────
% Args: #1=logo file  #2=job title  #3=company+location
%       #4=date range  #5=company URL (or {} for none)
%       #6=logo height (e.g. 0.50cm or 0.65cm)
\newcommand{\jobheader}[6]{%
  \vspace{3pt}%
  \def\joburlarg{#5}%
  \noindent
  \begin{minipage}[t]{0.055\textwidth}
    \vspace{0pt}
    \ifx\joburlarg\empty
      \includegraphics[width=\linewidth, height=#6,
                       keepaspectratio]{#1}%
    \else
      \href{#5}{\includegraphics[width=\linewidth, height=#6,
                       keepaspectratio]{#1}}%
    \fi
  \end{minipage}%
  \hspace{4pt}%
  \begin{minipage}[t]{0.73\textwidth}
    \vspace{0pt}
    {\fontsize{10}{11}\selectfont\textbf{#2}}\\[-3pt]
    {\fontsize{8.5}{9}\selectfont
      \ifx\joburlarg\empty
        \textcolor{accent}{#3}%
      \else
        \href{#5}{\textcolor{accent}{#3}}%
      \fi
    }
  \end{minipage}%
  \hfill
  \begin{minipage}[t]{0.175\textwidth}
    \vspace{1pt}
    \raggedleft
    {\fontsize{8.5}{9}\selectfont\textcolor{graydim}{#4}}
  \end{minipage}%
  \vspace{0pt}%
}

% ── Recognition / leadership entry ─────────────────────────
% Args: #1=logo file  #2=title  #3=org + date string
%       #4=URL (or {} for none)
\newcommand{\recogentry}[4]{%
  \vspace{1pt}%
  \def\recourlarg{#4}%
  \noindent
  \begin{minipage}[t]{0.045\textwidth}
    \vspace{1pt}
    \ifx\recourlarg\empty
      \includegraphics[width=\linewidth, height=0.38cm,
                       keepaspectratio]{#1}%
    \else
      \href{#4}{\includegraphics[width=\linewidth, height=0.38cm,
                       keepaspectratio]{#1}}%
    \fi
  \end{minipage}%
  \hspace{4pt}%
  \begin{minipage}[t]{0.91\textwidth}
    \vspace{0pt}
    {\fontsize{9.5}{11}\selectfont
      \ifx\recourlarg\empty
        \textbf{#2}\enspace\textcolor{accent}{#3}%
      \else
        \href{#4}{\textbf{#2}}\enspace\textcolor{accent}{#3}%
      \fi
    }
  \end{minipage}%
}
